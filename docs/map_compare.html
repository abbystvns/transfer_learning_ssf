nobs = 50
p = 100
#nsim = 1000

errors = []
for i in range(10):
    for k in [1, 5, 10, 25, 50]:
        for p in [100]:
            nsim = k*nobs
            theta = np.concatenate([np.ones(10), -1*np.ones(10), np.zeros(p-20)]) 
            X_obs, y_obs, X_sim, y_sim, cov = generate_simobs(nobs, nsim, .9, theta, sigma=.1)

            #obs only good init
            theta_hat, cov_hat = altmin(X_obs, y_obs, nobs, 1, init='good')
            errors.append([k, p, 'Obs only (good init)', np.linalg.norm(theta-theta_hat)])

            # init with obs 
            rig = linear_model.RidgeCV()
            rig.fit(X_obs, y_obs)
            theta_init = rig.coef_
            theta_hat, cov_hat = altmin(X_sim, y_sim, nsim, 1, init=theta_init)
            errors.append([k, p, 'Sims only (init with obs)', np.linalg.norm(theta-theta_hat)])
            
            # MAP estimator 
            chol = np.linalg.cholesky(cov)
            inv_ct = np.linalg.inv(chol)
            theta_hat = map_estimator(X_obs, y_obs, X_sim, y_sim, cov, .1)
            errors.append([k, p, 'MAP estimator', np.linalg.norm(theta-theta_hat)])
            
            X, y = stack_data(X_obs, X_sim, y_obs, y_sim)
            
            # obs only (not altmin)
            rig = linear_model.RidgeCV()
            theta_hat = rig.fit(X_obs, y_obs).coef_
            errors.append([k, p, 'Obs only ridge', np.linalg.norm(theta-theta_hat)])
            
            # obs + sim (not altmin)
            rig = linear_model.RidgeCV()
            theta_hat = rig.fit(X, y).coef_
            errors.append([k, p, 'Obs + sim ridge', np.linalg.norm(theta-theta_hat)])
                        
            # sims + obs (good init)
            theta_hat, cov_hat = altmin(X, y, nsim+nobs, 1, init='good')
            errors.append([k, p, 'Obs and sim (good init)', np.linalg.norm(theta-theta_hat)])
            
            # sims + obs (init with obs)
            rig = linear_model.RidgeCV()
            rig.fit(X_obs, y_obs)
            theta_init = rig.coef_
            theta_hat, cov_hat = altmin(X, y, nsim+nobs, 1, init=theta_init)
            errors.append([k, p, 'Obs and sim (init w/ obs)', np.linalg.norm(theta-theta_hat)])
                        
            # stacked sims + obs (good init)
            X, y = stack_data(X_obs, X_sim, y_obs, y_sim, how='k')
            theta_hat, cov_hat = altmin(X, y, int(nsim/nobs)+1, nobs, init='good')
            errors.append([k, p, 'Grouped obs + sim (good init)', np.linalg.norm(theta-theta_hat)])
