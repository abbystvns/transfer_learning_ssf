def altmin_switch(X, y, m, n, cov, theta, init='good', niter=10):
    errors = []
    mod = linear_model.RidgeCV()
    mod.fit(X,y)
    if init == 'good':
        theta_new = mod.coef_
    elif type(init)!=str:
        theta_new = init
    else:
        theta_new = np.random.normal(size=X.shape[1])
    rig = linear_model.Ridge(alpha=mod.alpha_)

    cov_hat = cov
    for i in range(niter):
        chol = np.linalg.cholesky(cov_hat + np.identity(m)*1e-6) #perturb so pd
        inv_ct = np.linalg.inv(chol)
        inv_ct_diag = np.kron(np.eye(int(n),dtype=int),inv_ct)
        theta_hat = mod.fit(inv_ct_diag@X, inv_ct_diag@y).coef_
        errors.append(np.linalg.norm(theta-theta_hat))

        cov_hat = sum([np.outer((y[j:j+m] - X[j:j+m]@theta_hat),(y[j:j+m] - X[j:j+m]@theta_hat) ) for j in np.arange(0, n*m, m)])/n # sample covariance of residuals
        # MAP estimator using cov_hat
        #chol = np.linalg.cholesky(cov_hat)
    #print('did not converge, error is: ', np.linalg.norm(theta_hat - theta_new))
    return theta_hat, cov_hat, errors

cov_errors = []
errors = []
sigma=.1

for i in range(10):
    for k in [1, 5, 10, 25, 50]:
        for p in [100]:
            nsim = k*nobs
            theta = np.concatenate([np.ones(10), -1*np.ones(10), np.zeros(p-20)])
            X_obs, y_obs, X_sim, y_sim, cov = generate_simobs(nobs, nsim, .9, theta, sigma=sigma)
            Sigma = scipy.linalg.block_diag(sigma**2*np.eye(nobs), X_sim@cov@X_sim.T+sigma**2*np.eye(nsim))
            X, y = stack_data(X_obs, X_sim, y_obs, y_sim)

            # obs only (not altmin)
            rig = linear_model.RidgeCV()
            theta_hat = rig.fit(X_obs, y_obs).coef_
            errors.append([k, p, 'Obs only ridge', np.linalg.norm(theta-theta_hat)])

            # obs + sim (not altmin)
            rig = linear_model.RidgeCV()
            theta_hat = rig.fit(X, y).coef_
            errors.append([k, p, 'Obs + sim ridge', np.linalg.norm(theta-theta_hat)])

            # MAP estimator 
            chol = np.linalg.cholesky(cov)
            inv_ct = np.linalg.inv(chol)
            theta_hat = map_estimator(X_obs, y_obs, X_sim, y_sim, cov, .1)
            errors.append([k, p, 'MAP estimator', np.linalg.norm(theta-theta_hat)])

            # sims + obs (init with obs)
            rig = linear_model.RidgeCV()
            rig.fit(X_obs, y_obs)
            theta_init = rig.coef_
